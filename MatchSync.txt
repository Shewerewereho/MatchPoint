-- 1. DATABASE SETUP
CREATE DATABASE MatchSyncDB;
GO
USE MatchSyncDB;
GO

-- 2. USER ROLES (Super Admin, Staff, Customer)
CREATE TABLE Roles (
    RoleID INT PRIMARY KEY, -- 0: Admin, 1: Staff, 2: Customer
    RoleName VARCHAR(20) NOT NULL
);
INSERT INTO Roles VALUES (0, 'Super Admin'), (1, 'Staff'), (2, 'Customer');

-- 3. USERS (Strict Account-Only Policy)
-- Supports OTP verification and Proven User logic[cite: 11, 13, 18, 21].
CREATE TABLE Users (
    UserID INT PRIMARY KEY IDENTITY(1,1),
    FullName VARCHAR(100) NOT NULL,
    Username VARCHAR(50) UNIQUE NOT NULL,
    PasswordHash VARCHAR(255) NOT NULL,
    PhoneNumber VARCHAR(15) NOT NULL,
    Email VARCHAR(100),
    RoleID INT DEFAULT 2, 
    IsVerified BIT DEFAULT 0, -- OTP status 
    IsProvenUser BIT DEFAULT 0, -- Unlocks after 5 qualifying bookings [cite: 13, 21]
    LoyaltyPoints INT DEFAULT 0,
    CONSTRAINT FK_UserRole FOREIGN KEY (RoleID) REFERENCES Roles(RoleID)
);

-- 4. FACILITY COURTS (5 Badminton, 1 Volleyball)
-- Supports the specific volleyball peak-hour pricing[cite: 17].
CREATE TABLE Courts (
    CourtID INT PRIMARY KEY IDENTITY(1,1),
    CourtName VARCHAR(50) NOT NULL, 
    CourtType VARCHAR(20) NOT NULL, -- 'Badminton' or 'Volleyball'
    BasePrice DECIMAL(10, 2) NOT NULL
);

-- 5. BOOKINGS & RESERVATIONS
-- Handles the 24-hour refund/reschedule window[cite: 10, 23].
CREATE TABLE Bookings (
    BookingID INT PRIMARY KEY IDENTITY(1,1),
    ReferenceCode AS ('MS-' + CAST(BookingID AS VARCHAR(10))), 
    UserID INT,
    CourtID INT,
    BookingDate DATE NOT NULL,
    StartTime TIME NOT NULL,
    EndTime TIME NOT NULL,
    BookingStatus VARCHAR(20) DEFAULT 'Pending', -- 'Pending', 'Confirmed', 'Ongoing', 'Completed', 'Cancelled'
    QRCodeToken UNIQUEIDENTIFIER DEFAULT NEWID(), -- For digital receipt and check-in 
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID),
    FOREIGN KEY (CourtID) REFERENCES Courts(CourtID)
);

-- 6. PAYMENTS (MongoPay Integration)
-- Tracks 50% downpayments[cite: 9, 19, 33].
CREATE TABLE Payments (
    PaymentID INT PRIMARY KEY IDENTITY(1,1),
    BookingID INT,
    MongoPayTransID VARCHAR(100),
    TotalAmount DECIMAL(10, 2),
    DownpaymentAmount DECIMAL(10, 2), -- 50% paid online 
    RemainingBalance DECIMAL(10, 2), -- Collected at POS [cite: 19]
    PaymentStatus VARCHAR(20), 
    FOREIGN KEY (BookingID) REFERENCES Bookings(BookingID)
);

-- 7. INVENTORY & SERVICES (Retail, Rentals, Stringing)
-- Includes "Services" category for racquet regutting[cite: 12, 20].
CREATE TABLE Inventory (
    ItemID INT PRIMARY KEY IDENTITY(1,1),
    ItemName VARCHAR(100) NOT NULL,
    Barcode VARCHAR(50) UNIQUE, -- For barcode scanning 
    Category VARCHAR(20), -- 'Retail', 'Rental', or 'Service'
    Price DECIMAL(10, 2) NOT NULL,
    StockQuantity INT DEFAULT 0
);

-- 8. STAFF-MANAGED STOCK UPDATES
-- Tracks which staff member added new stock for accountability.
CREATE TABLE StockUpdates (
    UpdateID INT PRIMARY KEY IDENTITY(1,1),
    ItemID INT,
    AddedQty INT,
    StaffUserID INT, 
    UpdateDate DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (ItemID) REFERENCES Inventory(ItemID),
    FOREIGN KEY (StaffUserID) REFERENCES Users(UserID)
);

-- 9. POS TRANSACTIONS (Includes Rental Tracking)
-- Monitors real-time status of rental gear.
CREATE TABLE POS_Transactions (
    TransID INT PRIMARY KEY IDENTITY(1,1),
    BookingID INT, 
    ItemID INT,
    Qty INT,
    SubTotal DECIMAL(10, 2),
    TransType VARCHAR(20), -- 'Sale', 'Rental', or 'Service'
    IsReturned BIT DEFAULT 0, -- For rental status tracking 
    ReturnDate DATETIME NULL,
    FOREIGN KEY (BookingID) REFERENCES Bookings(BookingID),
    FOREIGN KEY (ItemID) REFERENCES Inventory(ItemID)
);

-- 10. LOYALTY & AUDIT LOGS
CREATE TABLE LoyaltyLog (
    LogID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT,
    PointsChanged INT, 
    Reason VARCHAR(100), 
    LogDate DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID)
);

CREATE TABLE AdminActionLog (
    LogID INT PRIMARY KEY IDENTITY(1,1),
    AdminID INT,
    TargetUserID INT,
    ActionTaken VARCHAR(MAX), -- e.g., 'Changed Role to Staff'
    ActionDate DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (AdminID) REFERENCES Users(UserID)
);
